
CREATE TABLE tb_sequence_imagem (
                id_sequence serial NOT NULL,
                nm_sequence CHAR(1) NOT NULL,
                CONSTRAINT tb_sequence_imagem_pk PRIMARY KEY (id_sequence)
);


CREATE TABLE tb_imagem (
                id_imagem serial NOT NULL,
                id_sequence INTEGER NOT NULL,
                vl_url VARCHAR(100) NOT NULL,
                CONSTRAINT tb_imagem_pk PRIMARY KEY (id_imagem)
);


CREATE TABLE tb_tipo_avaliacao (
                id_avaliacao serial NOT NULL,
                nm_tipo_avaliacao VARCHAR(50) NOT NULL,
                fl_avaliador BOOLEAN,
                CONSTRAINT tb_tipo_avaliacao_pk PRIMARY KEY (id_avaliacao)
);


CREATE UNIQUE INDEX tb_tipo_avaliacao_idx
 ON tb_tipo_avaliacao
 ( nm_tipo_avaliacao );

CREATE TABLE tb_categoria (
                id_tag serial NOT NULL,
                nm_tag VARCHAR(100) NOT NULL,
                CONSTRAINT tb_categoria_pk PRIMARY KEY (id_tag)
);


CREATE UNIQUE INDEX tb_tags_idx
 ON tb_categoria

CREATE TABLE tb_usuario (
                id_usuario serial NOT NULL,
                nu_cpf CHAR(11) NOT NULL,
                nm_usuario VARCHAR(100) NOT NULL,
                vl_email VARCHAR(100) NOT NULL,
                nm_sobrenome VARCHAR NOT NULL,
                id_sequence INTEGER NOT NULL,
                vl_senha VARCHAR(100) NOT NULL,
                CONSTRAINT tb_usuario_pk PRIMARY KEY (id_usuario)
);


CREATE UNIQUE INDEX tb_usuario_idx
 ON tb_usuario
 ( nm_usuario, vl_email );

CREATE UNIQUE INDEX tb_usuario_idx1
 ON tb_usuario
 ( nu_cpf );

CREATE TABLE tb_grupo (
                id_grupo serial NOT NULL,
                id_usuario INTEGER NOT NULL,
                fl_grupo boolean NOT NULL,
                CONSTRAINT tb_grupo_pk PRIMARY KEY (id_grupo)
);


CREATE TABLE tb_grupo_conversa (
                id_conversa serial NOT NULL,
                id_grupo INTEGER NOT NULL,
                dt_envio DATE NOT NULL,
                hr_envio TIMESTAMP NOT NULL,
                vl_mensagem VARCHAR(1000) NOT NULL,
                CONSTRAINT tb_grupo_conversa_pk PRIMARY KEY (id_conversa)
);


CREATE TABLE tb_cardapio (
                id_cardapio serial NOT NULL,
                id_usuario INTEGER NOT NULL,
                dc_cardapio VARCHAR(100) NOT NULL,
                id_imagem INTEGER NOT NULL,
                CONSTRAINT tb_cardapio_pk PRIMARY KEY (id_cardapio)
);


CREATE TABLE tb_categoria_cardapio (
                id_cardapio INTEGER NOT NULL,

CREATE TABLE tb_local (
                id_local serial NOT NULL,
                nu_cep INTEGER NOT NULL,
                nu_casa INTEGER NOT NULL,
                id_usuario INTEGER NOT NULL,
                id_sequence INTEGER NOT NULL,
                nu_cnpj CHAR(14) NOT NULL,
                CONSTRAINT tb_local_pk PRIMARY KEY (id_local)
);


CREATE TABLE tb_refeicao (
                id_refeicao serial NOT NULL,
                id_cardapio INTEGER NOT NULL,
                id_local INTEGER NOT NULL,
                dt_agendamento DATE NOT NULL,
                hr_encontro TIMESTAMP NOT NULL,
                nu_max_convidados INTEGER NOT NULL,
                CONSTRAINT tb_refeicao_pk PRIMARY KEY (id_refeicao)
);


CREATE TABLE tb_encontro (
                id_encontro serial NOT NULL,
                id_refeicao INTEGER NOT NULL,
                fl_anfitriao_confirma BOOLEAN NOT NULL,
                id_sequence INTEGER NOT NULL,
                CONSTRAINT tb_encontro_pk PRIMARY KEY (id_encontro)
);


CREATE TABLE tb_encontro_usuario (
                id_encontro INTEGER NOT NULL,

                CONSTRAINT tb_encontro_usuario_pk PRIMARY KEY (id_encontro, id_usuario)
);


CREATE TABLE tb_avaliacao_encontro (
